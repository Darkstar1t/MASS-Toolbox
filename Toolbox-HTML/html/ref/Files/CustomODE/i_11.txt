modelOscil = model;
(*Make the compartment volume variables time dependent*)
\
setCustomRateLaws[modelOscil, 
 Thread[modelOscil[
    &quot;Fluxes&quot;] -&gt; (modelOscil[&quot;Rates&quot;] /. 
     Toolbox`parameter[&quot;Volume&quot;, &quot;c&quot;] -&gt; 
      Toolbox`parameter[&quot;Volume&quot;, &quot;c&quot;][t])]]
(*Remove the compartment volume parameter from the model&apos;s parameters*)
\
setParameters[modelOscil, FilterRules[modelOscil[&quot;Parameters&quot;], Except[
Toolbox`parameter[&quot;Volume&quot;, &quot;c&quot;]]]]
(*Set an initial condition for the time variable compartment volume*)
\
updateInitialConditions[modelOscil, {Toolbox`parameter[
   &quot;Volume&quot;, &quot;c&quot;] -&gt; 1 Liter}]
(*Add differential equation that describes the dynamics of the \
compartment volume*)
updateCustomODE[modelOscil, {Toolbox`parameter[
     &quot;Volume&quot;, &quot;c&quot;]&apos;[t] == 
   Sin[t .5] Toolbox`parameter[&quot;Volume&quot;, &quot;c&quot;][t]}]</pre><?xml version='1.0'?>
